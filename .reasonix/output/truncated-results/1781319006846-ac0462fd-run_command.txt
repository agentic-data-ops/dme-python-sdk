import re

with open('reference/dme-api-stormgmt.md', 'r', encoding='utf-8') as f:
    lines = f.readlines()

param_headers = []
for i, line in enumerate(lines):
    stripped = line.strip()
    if re.match(r'^\*{0,2}查询参数\*{0,2}\s*$', stripped) or \
       re.match(r'^- \*{0,2}请求body参数\*{0,2}', stripped) or \
       re.match(r'^- \*{0,2}响应参数\*{0,2}', stripped) or \
       re.match(r'^.*对象包含如下属性：?\s*$', stripped):
        param_headers.append(i)

def is_proper_table(lines, start_idx):
    for j in range(start_idx, min(start_idx+20, len(lines))):
        line = lines[j].strip()
        if line.startswith('| 参数'):
            return True
        if line.startswith('|') and '|' in line[1:]:
            if line.count('|') >= 2:
                return True
        if line.startswith('#'):
            return False
        if line == '':
            continue
    return False

# Categorize broken sections
cat1 = []  # 查询参数 with no table (or just '无')
cat2 = []  # 请求body参数 with broken table (no pipe separators)
cat3 = []  # 响应参数 with broken table
cat4 = []  # 对象包含如下属性 with broken table (text after colon on same line)
cat5 = []  # 参数 是否必选 lines without pipe at start of line (naked parameter tables)

for idx in param_headers:
    if is_proper_table(lines, idx+1):
        continue
    stripped = lines[idx].strip()
    ln = idx+1
    
    if '查询参数' in stripped:
        # Check what follows
        next_nonempty = ''
        for j in range(idx+1, min(idx+5, len(lines))):
            nxt = lines[j].strip()
            if nxt:
                next_nonempty = nxt[:80]
                break
        cat1.append((ln, stripped[:100], next_nonempty))
    elif '请求body参数' in stripped:
        # Check if params are concatenated inline
        has_inline_params = '是否' in stripped and '类型' in stripped and '值域' in stripped
        cat2.append((ln, stripped[:120], has_inline_params))
    elif '响应参数' in stripped:
        has_inline_params = '是否' in stripped and '类型' in stripped
        cat3.append((ln, stripped[:120], has_inline_params))
    elif '对象包含如下属性' in stripped:
        # Check if inline params follow
        has_inline = False
        for j in range(idx, min(idx+5, len(lines))):
            if '参数' in lines[j] and '是否' in lines[j] and '|' not in lines[j]:
                has_inline = True
                break
        cat4.append((ln, stripped[:120], has_inline))

# Also find naked '参数 是否必选' lines that aren't preceded by a recognized header
naked_params = []
for i, line in enumerate(lines):
    stripped = line.strip()
    if stripped.startswith('参数 是否') and '|' not in stripped:
        # Check if this isn't already captured
        already = False
        for idx in param_headers:
            if abs(idx - i) <= 3:
                already = True
                break
        if not already:
            naked_params.append((i+1, stripped[:120]))
